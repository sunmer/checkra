Usage Example:

import { initLogger } from 'advanced-frontend-logger';

// Initialize with custom options
const logger = initLogger({
  renderErrorLogDiv: true,
  errorLogDivId: 'custom-error-log',
  startCollapsed: false
});

// Use like regular console, but with enhanced features
console.log('Hello world');
console.info('This is some information');
console.warn('This is a warning');
console.error('This is an error', new Error('Something went wrong'));
